## Usage in R

```r
library(Seurat)

# Define marker sets
tcell_markers <- c("CD3D", "CD3E", "CD8A", "CD4")
bcell_markers <- c("MS4A1", "CD79A", "CD79B")
mono_markers <- c("CD14", "FCGR3A", "LYZ")

# Visualize markers
FeaturePlot(seurat_obj, features = tcell_markers)
DotPlot(seurat_obj, features = c(tcell_markers, bcell_markers, mono_markers))
VlnPlot(seurat_obj, features = tcell_markers)

# Find cells expressing specific markers
cd3_positive <- WhichCells(seurat_obj, expression = CD3D > 0)
```
